## C++ Implementation

```cpp
#include <bits/stdc++.h>
 
using namespace std;
 
using ll = long long;

ll fastpow(ll x, ll y,const ll MOD){
	ll res = 1;
	while (y>0){
		if (y&1) { //If y is odd
			res = (res*x)%MOD;
		}
		y>>=1; //Equivalent to y/=2 but more optimized
		x = (x*x)%MOD;
	}
	return res;
}
 
void solve(){
	int t; 
	cin >> t;
	while (t--){
		ll a,b,c;
		cin >> a >> b >> c;
		b = fastpow(b,c,1e9+6);
		cout << fastpow(a,b,1e9+7) << "\n";
	}	
	return;
}

int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	solve();
	return 0;
}
```
